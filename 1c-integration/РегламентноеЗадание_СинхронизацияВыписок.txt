//            из АСБО в 1С:Предприятие.
//
// Рекомендуемое расписание: каждые 15 минут (или раз в час в непиковое время).

Процедура ВыполнитьСинхронизацию() Экспорт

    ЗаписьЖурналаРегистрации(
        "АСБО.Синхронизация",
        УровеньЖурналаРегистрации.Информация,
        ,
        ,
        "Начало синхронизации выписок из АСБО");

    // --- 1. Получение списка обработанных платежей ---
    Платежи = ИнтеграцияАСБО.ПолучитьПлатежиДляЗагрузки();

    Если Платежи.Количество() = 0 Тогда
        ЗаписьЖурналаРегистрации(
            "АСБО.Синхронизация",
            УровеньЖурналаРегистрации.Информация,
            ,
            ,
            "Нет новых платежей для загрузки");
        Возврат;
    КонецЕсли;

    // --- 2. Определение организации (получатель платежей) ---
    // В реальной конфигурации — из константы или настройки.
    // ИНН организации: 7713699602 (ООО «Социальные услуги»)
    Организация = ПолучитьОрганизациюПоУмолчанию();

    Если Не ЗначениеЗаполнено(Организация) Тогда
        ЗаписьЖурналаРегистрации(
            "АСБО.Синхронизация",
            УровеньЖурналаРегистрации.Ошибка,
            ,
            ,
            "Не задана организация по умолчанию для синхронизации");
        Возврат;
    КонецЕсли;

    // --- 3. Статистика ---
    Всего = Платежи.Количество();
    Загружено = 0;
    Ошибок = 0;

    // --- 4. Последовательная обработка каждого платежа ---
    Для Каждого Платеж Из Платежи Цикл

        Попытка

            ИнтеграцияАСБО.ЗагрузитьПлатеж(Платеж, Организация);
            Загружено = Загружено + 1;

            // Отправка подтверждения в АСБО:
            // PUT /api/payments/:id со статусом "posted"

        Исключение

            Ошибок = Ошибок + 1;

            ЗаписьЖурналаРегистрации(
                "АСБО.Синхронизация",
                УровеньЖурналаРегистрации.Ошибка,
                ,
                ,
                "Ошибка загрузки платежа № " + Платеж.doc_number
                + ": " + ОписаниеОшибки());

        КонецПопытки;

    КонецЦикла;

    // --- 5. Итоговое журналирование ---
    ИнтеграцияАСБО.ЗаписатьЖурналОбмена(
        "ЗагрузкаПлатежей",
        "Импорт",
        "Синхронизация из АСБО",
        Всего,
        "Успешно",
        "Загружено: " + Загружено + ", ошибок: " + Ошибок
    );

    ЗаписьЖурналаРегистрации(
        "АСБО.Синхронизация",
        УровеньЖурналаРегистрации.Информация,
        ,
        ,
        "Синхронизация завершена. Всего: " + Всего
        + ", загружено: " + Загружено
        + ", ошибок: " + Ошибок);

КонецПроцедуры

Функция ПолучитьОрганизациюПоУмолчанию()

    // Поиск организации по ИНН (ООО «Социальные услуги»)
    Запрос = Новый Запрос;
    Запрос.Текст =
        "ВЫБРАТЬ ПЕРВЫЕ 1
        |   Организации.Ссылка КАК Ссылка
        |ИЗ
        |   Справочник.Организации КАК Организации
        |ГДЕ
        |   Организации.ИНН = &ИНН";

    Запрос.УстановитьПараметр("ИНН", "7713699602");
    Результат = Запрос.Выполнить();
    Выборка = Результат.Выбрать();

    Если Выборка.Следующий() Тогда
        Возврат Выборка.Ссылка;
    КонецЕсли;

    // Если организация не найдена — возвращаем любую
    Возврат Справочники.Организации.ПустаяСсылка();

КонецФункции
